    Quickstart

    We are going to start from a simple example with the scene data/nerf_synthetic/lego , where a 3D Gaussian Splatting model will be trained, tested, exported and visualized. You can find code from RadianceFieldStudio/examples/quickstart .

    1. Create Work Space

    As RadianceFieldStudio have been installed as a package, we start from an empty project. Just make a new directory named Quickstart anywhere as our work space.

    mkdir Quickstart
cd Quickstart

    2. Data Preparation

    Copy the download URL from NeRF example data. The URL should start with https://drive.usercontent.google.com/download? .

    Download & Unzip
    mkdir data
wget "https://drive.usercontent.google.com/download?..." -O data/example.zip
unzip data/example.zip -d data
rm data/example.zip

    You will get two scene now: data/nerf_llff_data/fern and data/nerf_synthetic/lego .

    from rfstudio.data import VanillaDataset
from rfstudio.engine.cli import cli_from_task
from rfstudio.engine.trainer import ExperimentDumper, TrainerConfig, VanillaTrainer
from rfstudio.model import GaussianSplatting
from rfstudio.model.components.renderer import GaussianRasterizer

std_trainer = VanillaTrainer(
    dataset=VanillaDataset(
        path=...,
    ),
    model=GaussianSplatting(
        renderer=GaussianRasterizer(background_color='black'),
        sh_degree=0
    ),
    experiment=ExperimentDumper(name=...),
    train=TrainerConfig(
        num_steps=30000,
        batch_size=1,
        lr=1e-3,
        eps=1e-15,
        lr_decay=0.01,
        max_norm=None,
        num_steps_per_display=100,
        num_steps_per_val=300,
        num_steps_per_save=6000,
        mixed_precision=False
    ),
    cuda=0,
    seed=1
)

if __name__ == '__main__':
    cli_from_task(std_trainer)

    Then, run the script in your terminal and you can watch progress from the terminal

    In the Terminal
    python train.py \
    --dataset.path data/nerf_synthetic/lego \
    --experiment.name testing

    4. Check Training Results

    After training for ~5min, the task will be done and you can use Ctrl+C to exit.

    The related training results will be dumped at outputs/testing/${timestamp} :

    • Checkpoints: Model weights are saved at outputs/testing/${timestamp}/ckpts .

    • Validation: novel views as well as their ground truth are rendered at outputs/testing/${timestamp}/dump/val .

    • Log: training details will be logged at outputs/testing/${timestamp}/log.txt .

    • Arguments: training arguments will be automatically generated at outputs/testing/${timestamp}/task.py . (1)

      1. It enables you to rerun an experiment with the completely same arguments, simply by:

        python outputs/testing/${timestamp}/task.py

    5.2 Start From Task

    The above script succeed in executing the main function run . Similarly, Radiance Field Studio abstracts any executable script into a Task class, which always has a run method as the entry point of the script. Users need to import the Task class as a base class and override its run method through inheritance:

    from rfstudio.engine.cli import cli_from_task
from rfstudio.engine.task import Task


class Tester(Task):
    def run(self) -> None:
        print('Hello World!')


if __name__ == '__main__':
    tester = Tester()
    cli_from_task(tester)

    $ python test.py

    from dataclasses import dataclass

from rfstudio.engine.cli import cli_from_task
from rfstudio.engine.task import Task


@dataclass                     # dataclass decorator is a must
class Tester(Task):

    num_a: int = ...           # ellipsis means "required"
    num_b: int = 6             # default value is 6
    num_c: int = ...           # ellipsis means "required"
    info: str = 'Hello World!' # default value is 'Hello World!'

    def run(self) -> None:
        print(self.info)       # use self.${field} to access any field
        print(f'{self.num_a + self.num_b + self.num_c = }')


if __name__ == '__main__':
    tester = Tester()
    cli_from_task(tester)

from rfstudio.engine.cli import cli_from_task
from rfstudio.engine.task import Task
from rfstudio.engine.task_components import ExperimentLoader
from rfstudio.engine.trainer import VanillaTrainer
from rfstudio.model import GaussianSplatting


@dataclass
class Tester(Task):

    experiment: ExperimentLoader = ExperimentLoader(dir=...)

    def run(self) -> None:
        trainer = self.experiment.load()
        assert isinstance(trainer, VanillaTrainer)
        model = trainer.model
        assert isinstance(model, GaussianSplatting)
        model.setup(self.device)
        model.load_checkpoint(trainer.ckpt_path)
        print(f'#Gaussians = {model.gaussians.shape[0]}')


if __name__ == '__main__':

    cli_from_task(Tester(cuda=0))

    $ python test.py --experiment.dir ./outputs/testing/${timestamp}
#Gaussians = 116590

    from dataclasses import dataclass

import numpy as np
import torch

from rfstudio.engine.cli import cli_from_task
from rfstudio.engine.task import Task
from rfstudio.engine.task_components import ExperimentLoader
from rfstudio.engine.trainer import VanillaTrainer
from rfstudio.model import GaussianSplatting, TrainInfo


@dataclass
class Tester(Task):

    experiment: ExperimentLoader = ExperimentLoader(dir=...)

    def run(self) -> None:
        trainer = self.experiment.load()
        assert isinstance(trainer, VanillaTrainer)
        model = trainer.model
        assert isinstance(model, GaussianSplatting)
        dataset = trainer.dataset
        model.setup(self.device)
        model.load_checkpoint(path=trainer.ckpt_path)

        with torch.no_grad():
            model.set_train_mode(False)
            losses = []
            psnrs = []
            for inputs, gt_outputs, indices, size in dataset.get_loader(
                split='test',
                device=self.device,
                batch_size=trainer.train.batch_size,
                shuffle=False,
                infinite=False,
                lazy=False
            ):
                train_info = TrainInfo(
                    is_training=False,
                    optimizers=None,
                    dump_path=None,
                    curr_step=None,
                    total_step=None,
                    dataset_indices=indices,
                    dataset_size=size
                )
                metrics = model.get_train_metrics(inputs, gt_outputs, train_info)
                losses.append(metrics['loss'].item())
                psnrs.append(metrics['psnr'].item())
            print("Eval: loss={:.5f} psnr={:.3f}".format(np.mean(losses), np.mean(psnrs)))

if __name__ == '__main__':

    cli_from_task(Tester(cuda=0))

    Run the script and the results are:

    Eval: loss=0.01009 psnr=33.638

    from dataclasses import dataclass

from rfstudio.engine.cli import cli_from_task
from rfstudio.engine.task import Task
from rfstudio.engine.task_components import ExperimentLoader
from rfstudio.engine.trainer import VanillaTrainer
from rfstudio.model import GaussianSplatting
from rfstudio.visualization import vis_3dgs


@dataclass
class Vis3DGS(Task):

    experiment: ExperimentLoader = ExperimentLoader(dir=...)

    def run(self) -> None:
        trainer = self.experiment.load()
        assert isinstance(trainer, VanillaTrainer)
        model = trainer.model
        assert isinstance(model, GaussianSplatting)
        model.setup(self.device)
        model.load_checkpoint(path=trainer.ckpt_path)
        vis_3dgs(model)


if __name__ == '__main__':
    cli_from_task(Vis3DGS())

    Then, you can visualize 3DGS in the interactive web viewer.
